Build the LinearRegression Model to predict price of ticket.


    class LinearRegression:
    def __init__(self, x, y): 
        self.x = x  
        self.y = y
    def ModelS (self):
        sum_x2 = sum(self.x)**2
        sum_xx = 0
        sum_xy = 0
        nx = len(self.x)
        for c,d in zip (self.x, self.y):
            sum_xx += c**2
            sum_xy += c*d
        sxx = sum_xx - (1/nx)*sum_x2
        ny = len(self.y)
        sxy = sum_xy - (1/ny)*sum(self.x)*sum(self.y)
        b = sxy/sxx
        avg_x = sum(self.x)/nx
        avg_y = sum(self.y)/ny
        a = avg_y - b*avg_x
        self.b = b
        self.a = a
    def predict(self, newbill):
        newtip = self.a + self.b*newbill
        return newtip
